 def sort012(self, arr):
        # code here
        n = len(arr)
        start = 0
        end = n - 1
        i = 0
        while i <= end:
            if arr[i] == 0:
                arr[start], arr[i] = arr[i], arr[start]
                start += 1
                i += 1
                
            elif arr[i] == 2:
                arr[end], arr[i] = arr[i], arr[end]
                end -= 1
        
            else:
                i += 1
